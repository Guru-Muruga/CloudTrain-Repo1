#1. Initialization of Git repo in a folder
git init


#2. Adding my default branch as "Main"
git branch -M main


#3. Adding public url for this project
git remote add origin https://github.com/Guru-Muruga/CloudTrain-Repo1.git


#4. Adding files in staging area
git add test-file1.txt
git add test-file2.txt 

git add .


#5. Committing the changes to Local Repo1
git commit -m "first commit"


#6. Pushing to public repo
git push -u origin main


#7: Authentication fails, run below command - globally setting the github username, and login email
git config --global user.name "Your Name"
git config --global user.email "your_email@example.com"
